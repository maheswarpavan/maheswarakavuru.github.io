/**
 * Cross-subsidiary PO validation — User Event Script
 * Blocks invalid vendor-subsidiary combinations before DB write
 * Entry point: beforeSubmit
 */
define(['N/record', 'N/search', 'N/log'], function(record, search, log) {

  function beforeSubmit(ctx) {
    var allowed = [
      ctx.UserEventType.CREATE,
      ctx.UserEventType.EDIT
    ];
    if (allowed.indexOf(ctx.type) === -1) return;

    var rec      = ctx.newRecord;
    var subId    = rec.getValue({ fieldId: 'subsidiary' });
    var vendorId = rec.getValue({ fieldId: 'entity' });

    if (!isApprovedPair(subId, vendorId)) {
      log.error({
        title:   'PO Validation Failed',
        details: 'Vendor ' + vendorId + ' not approved for subsidiary ' + subId
      });
      throw new Error('Vendor is not approved for this subsidiary.');
    }
  }

  function isApprovedPair(subId, vendorId) {
    var results = search.create({
      type:    'customrecord_vendor_sub_matrix',
      filters: [
        ['custrecord_subsidiary', 'anyof', subId],
        'AND',
        ['custrecord_vendor', 'anyof', vendorId]
      ]
    }).run().getRange({ start: 0, end: 1 });
    return results.length > 0;
  }

  return { beforeSubmit: beforeSubmit };
});
/**
 * Nightly inventory reconciliation — Map/Reduce Script
 * Aggregates available quantity by location across 5,000+ items
 * Governance-compliant: search object passed as input
 */
define(['N/search', 'N/log'], function(search, log) {

  function getInputData() {
    return search.create({
      type:    search.Type.INVENTORY_ITEM,
      filters: [['isinactive', 'is', 'F']],
      columns: ['itemid', 'quantityavailable', 'location']
    });
  }

  function map(ctx) {
    var val = JSON.parse(ctx.value);
    var qty = parseFloat(val.values.quantityavailable) || 0;
    var loc = val.values.location.text || 'Unknown';
    ctx.write({ key: loc, value: qty });
  }

  function reduce(ctx) {
    var total = ctx.values.reduce(function(sum, v) {
      return sum + parseFloat(v);
    }, 0);
    ctx.write({ key: ctx.key, value: total.toFixed(2) });
  }

  function summarize(ctx) {
    ctx.output.iterator().each(function(k, v) {
      log.audit({ title: k, details: 'Total available: ' + v });
      return true;
    });
  }

  return { getInputData: getInputData, map: map,
           reduce: reduce, summarize: summarize };
});
/**
 * Custom inventory snapshot — RESTlet
 * Returns bin-level available qty across locations in one call
 * Called by external dashboard every 5 minutes via TBA
 */
define(['N/search', 'N/log'], function(search, log) {

  function get(params) {
    var locationFilter = params.location
      ? [['location', 'anyof', params.location]]
      : [];

    var results = [];

    search.create({
      type:    'inventoryitem',
      filters: [['quantityavailable', 'greaterthan', '0']]
               .concat(locationFilter),
      columns: ['itemid', 'quantityavailable',
                'location', 'binonhand']
    }).run().each(function(r) {
      results.push({
        sku:      r.getValue('itemid'),
        qty:      parseFloat(r.getValue('quantityavailable')),
        location: r.getText('location'),
        bin:      r.getText('binonhand') || '—'
      });
      return true;
    });

    return {
      status:    'ok',
      timestamp: new Date().toISOString(),
      count:     results.length,
      data:      results
    };
  }

  return { get: get };
});
